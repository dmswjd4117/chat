1. async await
``` js

function sleep(ms) {
    return new Promise(resolve => setTimeout(resolve, ms));
}

async function demo() {
    console.log("2")
    await sleep()
    console.log("4")
}

// await 는 프라미스가 처리될때까지 기다린다. 그 결과는 이후 반환된다.

console.log("1")
demo().then(()=>{
    console.log("5")
})

```
